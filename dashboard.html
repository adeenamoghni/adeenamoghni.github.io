data("rest_inspec")

inspection_df = 
  rest_inspec %>%
  na.omit() %>% 
  separate(inspection_date, into = c("year", "month", "day"), sep = "-") %>%
  filter (year == 2017) 
  
score_per_month = 
  inspection_df %>% 
  na.omit(inspection_df) %>% 
  group_by(month, boro) %>% 
  summarize(score = score)
